Description

Service for positioning and managing overlays (dropdowns, popovers, tooltips). Provides automatic repositioning on scroll/resize and collision detection.

Example :
export class DropdownComponent {
  private overlayService = inject(OverlayService);


  showDropdown(): void {
    const position = this.overlayService.calculatePosition(
      this.triggerElement,
      this.dropdownElement,
      { placement: 'bottom', autoReposition: true }
    );
    this.overlayService.applyPosition(this.dropdownElement, position);
  }
}
